{{/* getDate: Retrieve published date and, if modified, updated date. */}} {{- $published := .Date -}} {{- $updated := .Lastmod -}} {{/* Creating a random number of seconds+millisecs to ensure no two dates are exactly the same, required for feed validation. */}} {{- $title := partial "function/vars/getTitle" . -}} {{- $seed := int ( printf "0x%.8s" (md5 ($title.linked))) -}} {{- $rnd := mod (add (mul 13 $seed) 97) 400 -}} {{- $publishedTime := $published.Unix -}} {{- $publishedTime = add $publishedTime $rnd -}} {{- $publishedTime = $publishedTime | time.AsTime -}} {{- $updatedTime := $updated -}} {{- if eq .Date .Lastmod -}} {{- $updated = "" -}} {{- $updatedTime = "" -}} {{- else -}} {{- $updatedTime = .Lastmod.Unix -}} {{- $updatedTime = add $updatedTime $rnd -}} {{- $updatedTime = $updatedTime | time.AsTime -}} {{- end -}} {{/* Because we only use dates, with no times, listings in e.g. feeds come out as 12 midnight. This adds 12 hours to each date to at least have them listed as 12 noon. */}} {{- $publishedTime = ($publishedTime.Add (time.ParseDuration "12h")).Format "2006-01-02T15:04:05-07:00" -}} {{- with $updatedTime -}} {{- $updatedTime = ($updatedTime.Add (time.ParseDuration "12h")).Format "2006-01-02T15:04:05-07:00" -}} {{- end -}} {{- $updatedRSS := $updatedTime -}} {{- $publishedRSS := dateFormat "Mon, 02 Jan 2006 15:04:00 -0700" $publishedTime -}} {{- with $updatedTime -}} {{- $updatedRSS = dateFormat "Mon, 02 Jan 2006 15:04:00 -0700" $updatedTime -}} {{- end -}} {{- $date := dict "published" $published "updated" $updated "publishedTime" $publishedTime "updatedTime" $updatedTime "publishedAtom" $publishedTime "updatedAtom" $updatedTime "publishedRSS" $publishedRSS "updatedRSS" $updatedRSS -}} {{- return $date -}}
